276  size_t LockingHashTable<type>::table_length(size_t buffer_size) const
277  {
278  // The closest power of two small enough to fit in the buffer
279  size_t result = buffer_size / sizeof(struct table_entry_s);
280  if (result) {
281  result--;
282  result |= result >> 1;
283  result |= result >> 2;
284  result |= result >> 4;
285  result |= result >> 8;
286  result |= result >> 16;
287  result |= result >> 32;
288  result++;
289  result = result >> 1;
290  }
291  if (result == 0) {
292  throw Exception("LockingHashTable: Failing to created empty table, increase size", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
293  }
294  return result;
295  }

305  {
306  if (key == 0) {
307  throw Exception("LockingHashTable::insert(): zero is not a valid key", GEOPM_ERROR_INVALID, __FILE__, __LINE__);
308  }
309  size_t table_idx = hash(key);
310  int err = pthread_mutex_lock(&(m_table[table_idx].lock));
311  if (err) {
312  throw Exception("LockingHashTable::insert(): pthread_mutex_lock()", err, __FILE__, __LINE__);
313  }
314  bool is_stored = false;
315  for (size_t i = 0; !is_stored && i != GEOPM_HASH_TABLE_DEPTH_MAX; ++i) {
316  if (m_table[table_idx].key[i] == 0 ||
317  (m_table[table_idx].key[i] == key &&
318  !sticky(m_table[table_idx].value[i]))) {
319  m_table[table_idx].key[i] = key;
320  m_table[table_idx].value[i] = value;
321  is_stored = true;
322  }
323  }
324  err = pthread_mutex_unlock(&(m_table[table_idx].lock));
325  if (err) {
326  throw Exception("LockingHashTable::insert(): pthread_mutex_unlock()", err, __FILE__, __LINE__);
327  }

362  uint64_t LockingHashTable<type>::key(const std::string &name)
363  {
364  uint64_t result = 0;
365  int err = pthread_mutex_lock(&(m_key_map_lock));
366  if (err) {
367  throw Exception("LockingHashTable::key(): pthread_mutex_lock()", err, __FILE__, __LINE__);
368  }
369  auto key_map_it = m_key_map.find(name);
370  err = pthread_mutex_unlock(&(m_key_map_lock));
371  if (err) {
372  throw Exception("LockingHashTable::key(): pthread_mutex_unlock()", err, __FILE__, __LINE__);
373  }
374 
375  if (key_map_it != m_key_map.end()) {
376  result = key_map_it->second;
377  }
378  else {
379  result = geopm_crc32_str(0, (char *)(&name.front()));
380  if (!result) {
381  throw Exception("LockingHashTable::key(): CRC 32 hashed to zero!", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
382  }
383  err = pthread_mutex_lock(&(m_key_map_lock));
384  if (err) {
385  throw Exception("LockingHashTable::key(): pthread_mutex_lock()", err, __FILE__, __LINE__);
386  }
387  if (m_key_set.find(result) != m_key_set.end()) {
388  throw Exception("LockingHashTable::key(): String hash collision", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
389  }
390  m_key_set.insert(result);
391  m_key_map.insert(std::pair<const std::string, uint64_t>(name, result));
392  m_key_map_last = m_key_map.begin();
393  err = pthread_mutex_unlock(&(m_key_map_lock));
394  if (err) {
395  throw Exception("LockingHashTable::key(): pthread_mutex_unlock()", err, __FILE__, __LINE__);
396  }

458  while (m_key_map_last != m_key_map.end() &&
459  buffer_remain > (*m_key_map_last).first.length()) {
460  strncpy(buffer_ptr, (*m_key_map_last).first.c_str(), buffer_remain);
461  buffer_remain -= (*m_key_map_last).first.length() + 1;
462  buffer_ptr += (*m_key_map_last).first.length() + 1;
463  ++m_key_map_last;
464  }
465  memset(buffer_ptr, 0, buffer_remain);
466  if (m_key_map_last == m_key_map.end() && buffer_remain) {
467  // We are done, set last character to -1
468  buffer_ptr[buffer_remain] = (char) 1;
469  m_key_map_last = m_key_map.begin();
470  result = true;
471  }
472  else {
